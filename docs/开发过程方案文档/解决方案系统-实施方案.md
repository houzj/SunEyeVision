### 2.1 创建 StartupDecisionService

**文件路径**: `src/Workflow/StartupDecisionService.cs`

```csharp
using System;
using System.IO;

namespace SunEyeVision.Workflow
{
    /// <summary>
    /// 启动决策服务
    /// </summary>
    /// <remarks>
    /// 决定启动时是否显示配置界面
    /// </remarks>
    public class StartupDecisionService
    {
        private readonly ProjectManager _projectManager;

        public StartupDecisionService(ProjectManager projectManager)
        {
            _projectManager = projectManager ?? throw new ArgumentNullException(nameof(projectManager));
        }

        /// <summary>
        /// 获取启动决策
        /// </summary>
        /// <returns>启动决策</returns>
        public StartupDecision GetStartupDecision()
        {
            var runtimeConfig = _projectManager.RuntimeConfig;

            // 检查是否跳过配置界面
            if (runtimeConfig.SkipStartupConfig)
            {
                return StartupDecision.SkipConfiguration;
            }

            // 检查是否有最近使用的项目
            if (string.IsNullOrEmpty(runtimeConfig.CurrentProject))
            {
                // 没有最近项目，显示配置界面（空状态）
                return StartupDecision.ShowConfigurationWithEmptyState;
            }

            // 有最近项目，显示配置界面（有项目）
            return StartupDecision.ShowConfigurationWithRecentProject;
        }
    }

    /// <summary>
    /// 启动决策枚举
    /// </summary>
    public enum StartupDecision
    {
        /// <summary>
        /// 显示配置界面（空状态，无项目）
        /// </summary>
        ShowConfigurationWithEmptyState,

        /// <summary>
        /// 显示配置界面（有最近项目）
        /// </summary>
        ShowConfigurationWithRecentProject,

        /// <summary>
        /// 显示配置界面（默认）
        /// </summary>
        ShowConfiguration,

        /// <summary>
        /// 跳过配置界面，直接启动
        /// </summary>
        SkipConfiguration
    }
}
```

### 2.3 创建 SolutionConfigurationDialogViewModel

**文件路径**: `src/UI/ViewModels/SolutionConfigurationDialogViewModel.cs`

```csharp
using System;
using System.Collections.ObjectModel;
using System.Linq;
using System.Windows.Input;
using SunEyeVision.Workflow;
using SunEyeVision.Plugin.SDK.Logging;

namespace SunEyeVision.UI.ViewModels
{
    /// <summary>
    /// 解决方案配置对话框ViewModel
    /// </summary>
    public class SolutionConfigurationDialogViewModel : ViewModelBase
    {
        private readonly ProjectManager _projectManager;
        private readonly ILogger _logger;
        
        private Project? _selectedProject;
        private InspectionRecipe? _selectedRecipe;

        /// <summary>
        /// 项目列表
        /// </summary>
        public ObservableCollection<Project> Projects { get; }

        /// <summary>
        /// 当前选中项目
        /// </summary>
        public Project? SelectedProject
        {
            get => _selectedProject;
            set
            {
                if (SetProperty(ref _selectedProject, value))
                {
                    // 项目切换时，选中第一个配方
                    SelectedRecipe = value?.Recipes.FirstOrDefault();
                }
            }
        }

        /// <summary>
        /// 当前选中配方
        /// </summary>
        public InspectionRecipe? SelectedRecipe
        {
            get => _selectedRecipe;
            set => SetProperty(ref _selectedRecipe, value);
        }

        /// <summary>
        /// 启动时跳过配置界面
        /// </summary>
        public bool SkipStartupConfig
        {
            get => _projectManager.RuntimeConfig.SkipStartupConfig;
            set
            {
                _projectManager.RuntimeConfig.SkipStartupConfig = value;
                _projectManager.SaveRuntimeConfig();
                OnPropertyChanged();
            }
        }

        /// <summary>
        /// 是否可以启动
        /// </summary>
        public bool CanLaunch => SelectedProject != null && SelectedRecipe != null;

        /// <summary>
        /// 新建项目命令
        /// </summary>
        public ICommand NewProjectCommand { get; }

        /// <summary>
        /// 打开项目命令
        /// </summary>
        public ICommand OpenProjectCommand { get; }

        /// <summary>
        /// 删除项目命令
        /// </summary>
        public ICommand DeleteProjectCommand { get; }

        /// <summary>
        /// 添加配方命令
        /// </summary>
        public ICommand AddRecipeCommand { get; }

        /// <summary>
        /// 复制配方命令
        /// </summary>
        public ICommand DuplicateRecipeCommand { get; }

        /// <summary>
        /// 删除配方命令
        /// </summary>
        public ICommand DeleteRecipeCommand { get; }

        /// <summary>
        /// 跳过配置命令
        /// </summary>
        public ICommand SkipConfigurationCommand { get; }

        /// <summary>
        /// 启动命令
        /// </summary>
        public ICommand LaunchCommand { get; }

        /// <summary>
        /// 启动完成事件
        /// </summary>
        public event EventHandler? LaunchCompleted;

        /// <summary>
        /// 跳过配置事件
        /// </summary>
        public event EventHandler? SkipConfiguration;

        public SolutionConfigurationDialogViewModel(ProjectManager projectManager)
        {
            _projectManager = projectManager ?? throw new ArgumentNullException(nameof(projectManager));
            _logger = VisionLogger.Instance;

            Projects = new ObservableCollection<Project>(
                _projectManager.GetAllProjects());

            // 选中最近使用的项目
            var recentProject = Projects.FirstOrDefault(p => p.Id == _projectManager.RuntimeConfig.CurrentProject);
            SelectedProject = recentProject ?? Projects.FirstOrDefault();

            // 初始化命令
            NewProjectCommand = new RelayCommand(ExecuteNewProject);
            OpenProjectCommand = new RelayCommand(ExecuteOpenProject);
            DeleteProjectCommand = new RelayCommand<Project>(ExecuteDeleteProject, CanDeleteProject);
            AddRecipeCommand = new RelayCommand(ExecuteAddRecipe, CanAddRecipe);
            DuplicateRecipeCommand = new RelayCommand<InspectionRecipe>(ExecuteDuplicateRecipe, CanDuplicateRecipe);
            DeleteRecipeCommand = new RelayCommand<InspectionRecipe>(ExecuteDeleteRecipe, CanDeleteRecipe);
            SkipConfigurationCommand = new RelayCommand(ExecuteSkipConfiguration);
            LaunchCommand = new RelayCommand(ExecuteLaunch, CanLaunchExecute);
        }

        /// <summary>
        /// 新建项目
        /// </summary>
        private void ExecuteNewProject()
        {
            var dialog = new NewProjectDialog();
            var result = dialog.ShowDialog();

            if (result == true && dialog.ProjectName != null)
            {
                var project = _projectManager.CreateProject(
                    dialog.ProjectName,
                    dialog.ProductCode ?? string.Empty,
                    dialog.Description ?? string.Empty);

                Projects.Add(project);
                SelectedProject = project;

                _logger.Log(LogLevel.Success, $"创建项目: {project.Name}", "SolutionConfiguration");
            }
        }

        /// <summary>
        /// 打开项目
        /// </summary>
        private void ExecuteOpenProject()
        {
            var dialog = new Microsoft.Win32.OpenFileDialog
            {
                Title = "打开项目",
                Filter = "项目文件 (*.json)|*.json|所有文件 (*.*)|*.*",
                InitialDirectory = Path.Combine(_projectManager.SolutionDirectory, "projects")
            };

            if (dialog.ShowDialog() == true)
            {
                // 这里需要实现从文件加载项目的逻辑
                // 暂时跳过
                _logger.Log(LogLevel.Info, $"打开项目: {dialog.FileName}", "SolutionConfiguration");
            }
        }

        /// <summary>
        /// 删除项目
        /// </summary>
        private void ExecuteDeleteProject(Project? project)
        {
            if (project == null)
                return;

            var result = System.Windows.MessageBox.Show(
                $"确定要删除项目 '{project.Name}' 吗？\n此操作不可恢复。",
                "确认删除",
                System.Windows.MessageBoxButton.YesNo,
                System.Windows.MessageBoxImage.Warning);

            if (result == System.Windows.MessageBoxResult.Yes)
            {
                _projectManager.DeleteProject(project.Id);
                Projects.Remove(project);

                if (SelectedProject == project)
                {
                    SelectedProject = Projects.FirstOrDefault();
                }

                _logger.Log(LogLevel.Warning, $"删除项目: {project.Name}", "SolutionConfiguration");
            }
        }

        /// <summary>
        /// 添加配方
        /// </summary>
        private void ExecuteAddRecipe()
        {
            if (SelectedProject == null)
                return;

            var dialog = new NewRecipeDialog();
            var result = dialog.ShowDialog();

            if (result == true && dialog.RecipeName != null)
            {
                var recipe = SelectedProject.AddRecipe(
                    dialog.RecipeName,
                    dialog.Description ?? string.Empty);

                _projectManager.SaveProject(SelectedProject);
                OnPropertyChanged(nameof(SelectedRecipe));

                _logger.Log(LogLevel.Success, $"添加配方: {recipe.Name}", "SolutionConfiguration");
            }
        }

        /// <summary>
        /// 复制配方
        /// </summary>
        private void ExecuteDuplicateRecipe(InspectionRecipe? recipe)
        {
            if (recipe == null || SelectedProject == null)
                return;

            var dialog = new NewRecipeDialog
            {
                RecipeName = recipe.Name + " (副本)",
                Description = recipe.Description
            };

            var result = dialog.ShowDialog();

            if (result == true && dialog.RecipeName != null)
            {
                var cloned = SelectedProject.DuplicateRecipe(recipe.Id, dialog.RecipeName);
                _projectManager.SaveProject(SelectedProject);
                OnPropertyChanged(nameof(SelectedRecipe));

                _logger.Log(LogLevel.Success, $"复制配方: {recipe.Name} -> {cloned.Name}", "SolutionConfiguration");
            }
        }

        /// <summary>
        /// 删除配方
        /// </summary>
        private void ExecuteDeleteRecipe(InspectionRecipe? recipe)
        {
            if (recipe == null || SelectedProject == null)
                return;

            var result = System.Windows.MessageBox.Show(
                $"确定要删除配方 '{recipe.Name}' 吗？\n此操作不可恢复。",
                "确认删除",
                System.Windows.MessageBoxButton.YesNo,
                System.Windows.MessageBoxImage.Warning);

            if (result == System.Windows.MessageBoxResult.Yes)
            {
                SelectedProject.RemoveRecipe(recipe.Id);
                _projectManager.SaveProject(SelectedProject);
                OnPropertyChanged(nameof(SelectedRecipe));

                _logger.Log(LogLevel.Warning, $"删除配方: {recipe.Name}", "SolutionConfiguration");
            }
        }

        /// <summary>
        /// 跳过配置
        /// </summary>
        private void ExecuteSkipConfiguration()
        {
            SkipConfiguration?.Invoke(this, EventArgs.Empty);
        }

        /// <summary>
        /// 启动
        /// </summary>
        private void ExecuteLaunch()
        {
            if (SelectedProject == null || SelectedRecipe == null)
                return;

            // 设置当前项目和配方
            _projectManager.SetCurrentProject(SelectedProject.Id, SelectedRecipe.Name);

            _logger.Log(LogLevel.Info, $"启动项目: {SelectedProject.Name}, 配方: {SelectedRecipe.Name}", "SolutionConfiguration");

            LaunchCompleted?.Invoke(this, EventArgs.Empty);
        }

        /// <summary>
        /// 判断是否可以删除项目
        /// </summary>
        private bool CanDeleteProject(Project? project)
        {
            return project != null && Projects.Count > 1;
        }

        /// <summary>
        /// 判断是否可以添加配方
        /// </summary>
        private bool CanAddRecipe()
        {
            return SelectedProject != null;
        }

        /// <summary>
        /// 判断是否可以复制配方
        /// </summary>
        private bool CanDuplicateRecipe(InspectionRecipe? recipe)
        {
            return recipe != null && SelectedProject != null;
        }

        /// <summary>
        /// 判断是否可以删除配方
        /// </summary>
        private bool CanDeleteRecipe(InspectionRecipe? recipe)
        {
            return recipe != null && SelectedProject != null && SelectedProject.Recipes.Count > 1;
        }

        /// <summary>
        /// 判断是否可以启动
        /// </summary>
        private bool CanLaunchExecute()
        {
            return CanLaunch;
        }
    }
}
```

```csharp
using System.Windows;
using SunEyeVision.UI.Views.Windows;
using SunEyeVision.Workflow;
using SunEyeVision.UI.ViewModels;

namespace SunEyeVision.UI
{
    public partial class App : Application
    {
        private ProjectManager? _projectManager;

        protected override void OnStartup(StartupEventArgs e)
        {
            base.OnStartup(e);
            
            try
            {
                // 初始化 ProjectManager
                var solutionDirectory = Path.Combine(
                    Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData),
                    "SunEyeVision",
                    "solutions");
                
                _projectManager = new ProjectManager(solutionDirectory);
                
                // 创建启动决策服务
                var startupDecisionService = new StartupDecisionService(_projectManager);
                var decision = startupDecisionService.GetStartupDecision();
                
                // 根据决策显示配置界面或直接启动
                if (decision == StartupDecisionService.StartupDecision.SkipConfiguration)
                {
                    // 跳过配置，直接启动主界面
                    LaunchMainWindow(null, null);
                }
                else
                {
                    // 显示配置界面
                    ShowConfigurationDialog();
                }
            }
            catch (Exception ex)
            {
                MessageBox.Show($"启动失败: {ex.Message}", "错误", MessageBoxButton.OK, MessageBoxImage.Error);
                Shutdown(1);
            }
        }
        
        /// <summary>
        /// 显示配置对话框
        /// </summary>
        private void ShowConfigurationDialog()
        {
            var configDialog = new SolutionConfigurationDialog(_projectManager!);
            
            if (configDialog.ShowDialog() == true)
            {
                // 用户点击[启动]，加载项目和配方
                var project = configDialog.CurrentProject;
                var recipe = configDialog.CurrentRecipe;
                
                LaunchMainWindow(project, recipe);
            }
            else
            {
                // 用户点击[跳过配置]，直接启动主界面
                LaunchMainWindow(null, null);
            }
        }
        
        /// <summary>
        /// 启动主界面
        /// </summary>
        private void LaunchMainWindow(Project? project, InspectionRecipe? recipe)
        {
            var mainWindow = new MainWindow();
            
            if (mainWindow.DataContext is MainWindowViewModel viewModel)
            {
                viewModel.InitializeFromProjectManager(_projectManager!, project, recipe);
            }
            
            mainWindow.Show();
        }
    }
}
```
